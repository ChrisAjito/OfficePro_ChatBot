/**
 * Lightweight CSV Parser resilient to missing values and extra spaces.
 * Supports basic quoting.
 */
function parseCSV(csvText) {
  const lines = csvText.split('\n').filter(line => line.trim() !== '');
  if (lines.length === 0) return [];

  const parseLine = (line) => {
    const cols = [];
    let curr = '';
    let inQuotes = false;
    for (let j = 0; j < line.length; j++) {
      const char = line[j];
      // Handle escaped quotes ("")
      if (char === '"' && line[j+1] === '"') {
        curr += '"';
        j++;
      } else if (char === '"') {
        inQuotes = !inQuotes;
      } else if (char === ',' && !inQuotes) {
        cols.push(curr);
        curr = '';
      } else {
        curr += char;
      }
    }
    cols.push(curr);
    return cols;
  };

  const headers = parseLine(lines[0]).map(h => h.trim());
  const data = [];

  for (let i = 1; i < lines.length; i++) {
    const cols = parseLine(lines[i]);
    const rowData = {};
    headers.forEach((header, index) => {
      rowData[header] = cols[index] ? cols[index].trim() : ''; 
    });
    data.push(rowData);
  }
  return data;
}

/**
 * Calls the Gemini API with exponential backoff for reliability.
 */
async function fetchGeminiContent(prompt, retries = 3) {
  const apiKey = ""; // Provided by the execution environment
  const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
  
  let delay = 1000;
  for (let i = 0; i < retries; i++) {
    try {
      const response = await fetch(url, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          contents: [{ parts: [{ text: prompt }] }],
          systemInstruction: { 
            parts: [{ text: "You are Ajito, an expert education data assistant. Provide brief, engaging, and professional insights based ONLY on the provided data context. Keep responses to 2-3 sentences max." }] 
          }
        })
      });
      if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);
      const data = await response.json();
      return data.candidates?.[0]?.content?.parts?.[0]?.text || "No insights generated.";
    } catch (error) {
      if (i === retries - 1) throw error;
      await new Promise(res => setTimeout(res, delay));
      delay *= 2; // Exponential backoff: 1s, 2s, 4s...
    }
  }
}

const TPL_TOWN = "Tell me about this Town: ";
const TPL_SCHOOL = "Tell me about this School: ";

function searchTowns(data, query) {
  const q = query.toLowerCase().trim();
  if (!q) return [];

  const exactMatches = [];
  const partialMatches = [];

  data.forEach(row => {
    const town = (row['Town'] || '').toLowerCase();
    const address = (row['Address'] || '').toLowerCase();

    if (town === q) {
      exactMatches.push(row);
    } else if (town.includes(q) || address.includes(q)) {
      partialMatches.push(row);
    }
  });

  return [...exactMatches, ...partialMatches];
}

function searchSchools(data, query) {
  const q = query.toLowerCase().trim();
  if (!q) return null;

  let bestMatch = null;
  let partialMatches = [];

  for (const row of data) {
    const name = (row['Organization Name'] || '').toLowerCase();
    if (name === q) {
      bestMatch = row; // Exact match found
      break;
    } else if (name.includes(q)) {
      partialMatches.push(row);
    }
  }

  return bestMatch || (partialMatches.length > 0 ? partialMatches[0] : null);
}

function TownAnalyzer({ townName, data }) {
  const [analysis, setAnalysis] = useState('');
  const [loading, setLoading] = useState(false);

  const generateAnalysis = async () => {
    setLoading(true);
    try {
      const schoolSummary = data.map(d => `${d['Organization Name']} (${d['Business Type']}, Size: ${d['Size Metric']})`).join('; ');
      const prompt = `Analyze the educational landscape in ${townName} based ONLY on these schools: ${schoolSummary}. Provide a brief, 2-3 sentence overview of the options available to a parent.`;
      const result = await fetchGeminiContent(prompt);
      setAnalysis(result);
    } catch (err) {
      setAnalysis('Analysis unavailable at this moment. Please try again later.');
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="mt-2 bg-red-50 p-3 rounded-lg border border-red-100 shadow-sm">
      {!analysis && !loading && (
         <button 
           onClick={generateAnalysis} 
           className="w-full text-sm font-semibold text-red-700 hover:text-red-800 flex items-center justify-center gap-2 py-1 transition-colors"
         >
           ✨ Analyze {townName} Schools
         </button>
      )}
      {loading && <div className="text-sm text-red-600 animate-pulse text-center font-medium">✨ Analyzing local education landscape...</div>}
      {analysis && (
         <div className="text-sm text-gray-800">
            <strong className="text-red-700 flex items-center gap-1 mb-1">✨ AI Landscape Insight</strong>
            <p className="leading-relaxed">{analysis}</p>
         </div>
      )}
    </div>
  );
}

            className="px-4 py-2 hover:bg-red-50 text-sm text-gray-800 cursor-pointer border-b border-gray-50 last:border-0 transition-colors"
          >
            {suggestion}
          </li>
        ))}
      </ul>
    </div>
  );
}

// ============================================================================
// --- src/components/AskAjitoWidget.jsx ---
// ============================================================================
function AskAjitoWidget({ csvData, isOpen, setIsOpen }) {
  const [messages, setMessages] = useState([
    { id: 1, sender: 'bot', type: 'text', text: 'Hello! I am Ajito. How can I help you today? Please select a template below to get started.' }
  ]);
  const [inputValue, setInputValue] = useState('');
  const [suggestions, setSuggestions] = useState([]);
  const [isMinimized, setIsMinimized] = useState(false);
  
  const messagesEndRef = useRef(null);
  const inputRef = useRef(null);

  const scrollToBottom = () => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  };

  useEffect(() => {
    scrollToBottom();
  }, [messages, isOpen, isMinimized]);

  const handleInputChange = (e) => {
    const val = e.target.value;
    setInputValue(val);

    if (val.startsWith(TPL_TOWN)) {
      const query = val.replace(TPL_TOWN, '').trim();
      if (query.length >= 2) {
        const matches = searchTowns(csvData, query);
        const uniqueTowns = [...new Set(matches.map(m => m.Town))].filter(Boolean);
        setSuggestions(uniqueTowns.slice(0, 5));
      } else {
        setSuggestions([]);
      }
    } else if (val.startsWith(TPL_SCHOOL)) {
      const query = val.replace(TPL_SCHOOL, '').trim();
      if (query.length >= 2) {
        const matches = csvData.filter(row => (row['Organization Name'] || '').toLowerCase().includes(query.toLowerCase()));
        const uniqueSchools = [...new Set(matches.map(m => m['Organization Name']))].filter(Boolean);
        setSuggestions(uniqueSchools.slice(0, 5));
      } else {
        setSuggestions([]);
      }
    } else {
      setSuggestions([]);
    }
  };

  const handleTemplateSelect = (templateText) => {
    setInputValue(templateText);
    setSuggestions([]);
    if (inputRef.current) {
      inputRef.current.focus();
    }
  };

  const handleSuggestionClick = (suggestion) => {
    let finalQuery = '';
    if (inputValue.startsWith(TPL_TOWN)) {
      finalQuery = `${TPL_TOWN}${suggestion}`;
    } else if (inputValue.startsWith(TPL_SCHOOL)) {
      finalQuery = `${TPL_SCHOOL}${suggestion}`;
    }
    
    setInputValue(finalQuery);
    setSuggestions([]);
    if (inputRef.current) {
      inputRef.current.focus();
    }
  };

  const handleSend = (e, forcedQuery = null) => {
    if (e) e.preventDefault();
    
    const queryToProcess = forcedQuery !== null ? forcedQuery : inputValue;
    if (!queryToProcess.trim()) return;

    const newUserMsg = { id: Date.now(), sender: 'user', type: 'text', text: queryToProcess };
    setMessages(prev => [...prev, newUserMsg]);
    setInputValue('');
    setSuggestions([]);

    setTimeout(() => {
      if (queryToProcess.startsWith(TPL_TOWN)) {
        const queryTerm = queryToProcess.replace(TPL_TOWN, '').trim();
        const results = searchTowns(csvData, queryTerm);
        
        if (results.length > 0) {
          setMessages(prev => [...prev, { 
            id: Date.now() + 1, 
            sender: 'bot', 
            type: 'town_results', 
            data: results,
            townName: queryTerm,
            text: `Here is what I found for "${queryTerm}":`
          }]);
        } else {
          setMessages(prev => [...prev, { id: Date.now() + 1, sender: 'bot', type: 'text', text: 'No matching town records found.' }]);
        }
      } 
      else if (queryToProcess.startsWith(TPL_SCHOOL)) {
        const queryTerm = queryToProcess.replace(TPL_SCHOOL, '').trim();
        const result = searchSchools(csvData, queryTerm);
        
        if (result) {
          setMessages(prev => [...prev, { 
            id: Date.now() + 1, 
            sender: 'bot', 
            type: 'school_result', 
            data: result,
            text: `Here are the details for ${result['Organization Name']}:`
          }]);
        } else {
          setMessages(prev => [...prev, { id: Date.now() + 1, sender: 'bot', type: 'text', text: 'No matching school records found.' }]);
        }
      } 
      else {
        setMessages(prev => [...prev, { 
          id: Date.now() + 1, 
          sender: 'bot', 
          type: 'text', 
          text: 'Please use the templates below to improve responses.' 
        }]);
      }
    }, 400);
  };

  const handleTownCardClick = (schoolName) => {
    const query = `${TPL_SCHOOL}${schoolName}`;
    handleSend(null, query);
  };

  if (!isOpen) {
    return (
      <button 
        onClick={() => setIsOpen(true)}
        className="fixed bottom-6 right-6 w-16 h-16 bg-red-600 text-white rounded-full flex items-center justify-center shadow-2xl hover:bg-red-700 hover:scale-105 transition-all z-50 group"
        aria-label="Open Ask Ajito"
      >
        <MessageIcon />
        <span className="absolute right-full mr-4 bg-black text-white px-3 py-1 rounded-md text-sm font-medium opacity-0 group-hover:opacity-100 transition-opacity whitespace-nowrap pointer-events-none shadow-lg">
          Ask Ajito
        </span>
      </button>
    );
  }

  return (
    <div className={`fixed right-0 sm:right-6 bottom-0 sm:bottom-6 w-full sm:w-[400px] bg-white sm:rounded-2xl shadow-2xl flex flex-col z-50 transition-all duration-300 border border-gray-200 ${isMinimized ? 'h-[60px]' : 'h-[100dvh] sm:h-[650px] max-h-[100dvh]'}`}>
      
      {/* Header */}
      <div className="bg-black text-white px-4 py-3 flex justify-between items-center sm:rounded-t-2xl flex-shrink-0 cursor-pointer" onClick={() => setIsMinimized(!isMinimized)}>
        <div className="flex items-center gap-2">
          <div className="w-8 h-8 bg-red-600 rounded-full flex items-center justify-center font-bold">A</div>
          <h2 className="font-bold text-lg tracking-wide">Ask Ajito</h2>
        </div>
        <div className="flex items-center gap-3">
          <button 
            onClick={(e) => { e.stopPropagation(); setIsMinimized(!isMinimized); }} 
            className="text-gray-300 hover:text-white transition-colors p-1"
          >
            <MinimizeIcon />
          </button>
          <button 
            onClick={(e) => { e.stopPropagation(); setIsOpen(false); }} 
            className="text-gray-300 hover:text-white transition-colors p-1"
          >
            <XIcon />
          </button>
        </div>
      </div>

      {/* Chat Area & Inputs (Hidden when minimized) */}
      {!isMinimized && (
        <>
          {/* Chat Window */}
          <div className="flex-1 overflow-y-auto p-4 bg-gray-50 flex flex-col gap-4">
            {messages.map((msg) => (
              <div key={msg.id} className={`flex ${msg.sender === 'user' ? 'justify-end' : 'justify-start'} w-full`}>
                <div className={`max-w-[85%] rounded-2xl p-3 shadow-sm ${
                  msg.sender === 'user' 
                    ? 'bg-red-600 text-white rounded-br-none' 
                    : 'bg-white text-black border border-gray-200 rounded-bl-none'
                }`}>
                  <p className="text-sm leading-relaxed whitespace-pre-wrap">{msg.text}</p>
                  
                  {msg.type === 'town_results' && (
                    <div className="mt-3 flex flex-col gap-2">
                      {msg.data.map((school, idx) => (
                        <ResultCard key={idx} data={school} onClick={handleTownCardClick} />
                      ))}
                      <TownAnalyzer townName={msg.townName} data={msg.data} />
                    </div>
                  )}

                  {msg.type === 'school_result' && (
                    <SchoolDetails data={msg.data} />
                  )}
                </div>
              </div>
            ))}
            <div ref={messagesEndRef} />
          </div>

          {/* Input Area */}
          <div className="flex-shrink-0 bg-white relative">
            <SuggestionList suggestions={suggestions} onSelect={handleSuggestionClick} />
            
            <form onSubmit={handleSend} className="p-3 flex items-end gap-2 border-t border-gray-100">
              <textarea
                ref={inputRef}
                value={inputValue}
                onChange={handleInputChange}
                onKeyDown={(e) => {
                  if (e.key === 'Enter' && !e.shiftKey) {
                    e.preventDefault();
                    handleSend(e);
                  }
                }}
                placeholder="Select a template or type..."
                className="flex-1 max-h-32 min-h-[44px] bg-gray-100 border-none rounded-xl px-4 py-3 text-sm text-black focus:ring-2 focus:ring-red-500 focus:outline-none resize-none"
                rows="1"
              />
              <button 
                type="submit" 
                disabled={!inputValue.trim()}
                className="w-11 h-11 flex-shrink-0 bg-red-600 text-white rounded-full flex items-center justify-center hover:bg-red-700 disabled:bg-gray-300 disabled:cursor-not-allowed transition-colors"
              >
                <SendIcon />
              </button>
            </form>
            
            <PromptTemplates onSelect={handleTemplateSelect} />
          </div>
        </>
      )}
    </div>
  );
}
